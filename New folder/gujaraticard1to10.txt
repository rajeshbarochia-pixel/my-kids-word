    <script>
        // અંકો માટે ઓડિયો ફાઇલ્સ
        const numberSounds = {
            1: "https://www.gujaratilexicon.com/resources/audio/dictionary/1.mp3",
            2: "https://www.gujaratilexicon.com/resources/audio/dictionary/2.mp3",
            3: "https://www.gujaratilexicon.com/resources/audio/dictionary/3.mp3",
            4: "https://www.gujaratilexicon.com/resources/audio/dictionary/4.mp3",
            5: "https://www.gujaratilexicon.com/resources/audio/dictionary/5.mp3",
            6: "https://www.gujaratilexicon.com/resources/audio/dictionary/6.mp3",
            7: "https://www.gujaratilexicon.com/resources/audio/dictionary/7.mp3",
            8: "https://www.gujaratilexicon.com/resources/audio/dictionary/8.mp3",
            9: "https://www.gujaratilexicon.com/resources/audio/dictionary/9.mp3",
            10: "https://www.gujaratilexicon.com/resources/audio/dictionary/10.mp3"
        };

        // Track currently playing audio to prevent overlapping
        let currentAudio = null;

        // અંક બોલવા માટેની ફંક્શન
        function playSound(number) {
            // Stop any currently playing audio
            if (currentAudio) {
                currentAudio.pause();
                currentAudio = null;
            }
            
            // જો ઓડિયો ફાઇલ ઉપલબ્ધ ન હોય તો ટેક્સ્ટ-ટુ-સ્પીચનો ઉપયોગ કરો
            if (!numberSounds[number] || !canPlayAudio(numberSounds[number])) {
                speakNumber(number);
                return;
            }
            
            const audio = new Audio(numberSounds[number]);
            currentAudio = audio;
            
            audio.play().catch(error => {
                console.error("ઓડિયો પ્લે કરવામાં ભૂલ:", error);
                speakNumber(number);
            });
            
            // When audio finishes, reset currentAudio
            audio.onended = function() {
                currentAudio = null;
            };
        }

        // ઓડિયો ફાઇલ ચકાસવા માટેની ફંક્શન
        function canPlayAudio(url) {
            const audio = new Audio();
            try {
                audio.src = url;
                return true;
            } catch (e) {
                return false;
            }
        }

        // ટેક્સ્ટ-ટુ-સ્પીચનો ઉપયોગ કરીને અંક બોલવા માટેની ફંક્શન
        function speakNumber(number) {
            if ('speechSynthesis' in window) {
                const gujaratiNumbers = {
                    1: "એક",
                    2: "બે",
                    3: "ત્રણ",
                    4: "ચાર",
                    5: "પાંચ",
                    6: "છ",
                    7: "સાત",
                    8: "આઠ",
                    9: "નવ",
                    10: "દસ"
                };
                
                // Stop any ongoing speech
                window.speechSynthesis.cancel();
                
                const utterance = new SpeechSynthesisUtterance(gujaratiNumbers[number]);
                utterance.lang = 'gu-IN';
                window.speechSynthesis.speak(utterance);
            }
        }
        
        // Prevent default touch behavior to avoid highlighting
        document.addEventListener('touchstart', function(e) {
            if (e.target.classList.contains('number-card')) {
                e.preventDefault();
            }
        }, { passive: false });
    </script>
