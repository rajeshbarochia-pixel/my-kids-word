  <script>
    const DIG = ["૧૧","૧૨","૧૩","૧૪","૧૫","૧૬","૧૭","૧૮","૧૯","૨૦"];
    const WORD = ["અગિયાર","બાર","તેર","ચૌદ","પંદર","સોળ","સત્તર","અઢાર","ઓગણીસ","વીસ"];
    const COLORS = [
      "#2196f3","#ffeb3b","#8bc34a","#ff5722","#9c27b0",
      "#3f51b5","#009688","#e91e63","#cddc39","#673ab7"
    ];

    const board = document.getElementById("board");
    const playAllBtn = document.getElementById("playAllBtn");
    const resetBtn = document.getElementById("resetBtn");

    function createCard(i){
      const card=document.createElement("div");
      card.className="card";
      const inner=document.createElement("div");
      inner.className="inner";
      const front=document.createElement("div");
      front.className="face front";
      front.style.setProperty("--clr", COLORS[i]);
      front.textContent=DIG[i];
      const back=document.createElement("div");
      back.className="face back";
      back.textContent=WORD[i];  // ફક્ત ગુજરાતી શબ્દ
      inner.append(front,back);
      card.appendChild(inner);

      card.addEventListener("click",()=>{
        card.classList.add("flipped");
        speak(WORD[i]);  // ફક્ત ગુજરાતી અવાજ
        setTimeout(()=>card.classList.remove("flipped"),2000);
      });

      return card;
    }

    function speak(txt){
      if(!("speechSynthesis" in window)) return;
      const u=new SpeechSynthesisUtterance(txt);
      u.lang="gu-IN";   // ગુજરાતી અવાજ
      u.rate=0.95;
      speechSynthesis.cancel();
      speechSynthesis.speak(u);
    }

    function playAll() {
      // Play all numbers in sequence with delays
      WORD.forEach((word, index) => {
        setTimeout(() => {
          speak(word);
        }, index * 2200); // 2.2 second delay between each
      });
    }

    function resetCards() {
      const cards = document.querySelectorAll('.card');
      cards.forEach(card => {
        card.classList.remove('flipped');
      });
    }

    // Initialize the board with cards
    DIG.forEach((_,i)=>board.appendChild(createCard(i)));

    // Add event listeners to buttons
    playAllBtn.addEventListener('click', playAll);
    resetBtn.addEventListener('click', resetCards);
  </script>
